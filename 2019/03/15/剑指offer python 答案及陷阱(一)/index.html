class TreeNode():
    def __init__(self,x):
            self.val = x
            self.right = None
            self.left = None
def hasSubtree (pRoot1,pRoot2):
        if not pRoot1 or not pRoot2:
                return False
        return is_Subtree(pRoot1,pRoot2) or \
                hasSubtree(pRoot1.right,pRoot2) or hasSubtree(pRoot1.left,pRoot2)
def is_Subtree (A, B):
        if not B:
                return True
        if not A or A.val != B.val:
                return False
        return is_Subtree(A.left,B.left) and is_Subtree(A.right,B.right)

root.left = TreeNode(3)

  • 有序链表合并(注意特殊情况,即代码的鲁棒性):

    class LinkNode():
    def __init__(self, x):
            self.val = x
            self.next = None
def merge(pHead1,pHead2):
        if not pHead1 and pHead2:
                return pHead2
        if not pHead2 and pHead1:
                return pHead1
        if not pHead2 and not pHead1:
                return False

        head = LinkNode(0) #随便一个值
        item = head
        while pHead1 and pHead2:
                if pHead1.val > pHead2.val:
                        item.next = pHead2
                        pHead2 = pHead2.next
                        item = item.next
                        item.next = None
                else:
                        item.next = pHead1
                        pHead1 = pHead1.next
                        item = item.next
                        item.next = None
        if pHead1 == None:
                item.next = pHead2
        else:
                item.next = pHead1
        return head.next

pHead1 = LinkNode(1)
pHead1.next = LinkNode(2)
pHead2 = LinkNode(2)
pHead2.next = LinkNode(4)
head = merge(pHead1,pHead2)
while head:
        print(head.val, end = ' ')
        head = head.next

  • 调整数组顺序奇数位于偶数前面(时间复杂度为O(n))

    def ReorderOddEven(array):
     high = len(array) - 1
     low = 0
     if high == -1:
             return  False
     while low < high:
             while low < high and array[low] % 2:
                     low += 1
             if low >= high:
                     break
             while low < high and array[high] % 2 == 0:
                     high -= 1
             if low < high:
                     array[low],array[high] = array[high],array[low]
                     low += 1
                     high -= 1
     return array
 array = [1,5,6,8,7,3]
 print(ReorderOddEven(array))

    def printNumber(n):
    number = ['0'] * n
    for i in range(10):
            number[0] = str(i)
            printIndex(number,n,0)
def printIndex(number,n,index):
        if index == n-1:
                printReal(number)
                return 
        else:
                for i in range(10):
                        number[index+1] = str(i)
                        printIndex(number,n,index+1)
def printReal(number):
        i = 0
        while i <= len(number)-1:
                if number[i] == '0':
                        i += 1
                else:
                        print(''.join(number[i:]))
                        break

    def fibonacci(n):
      f =[0,1,1]
      if n == 0:
          return 0
      if n == 1 or n == 2:
          return 1
      for i in range(n-1):
          f[2] = f[1]+ f[0]
          f[0] = f[1]
          f[1] = f[2]
      return f[1]

  • 重建二叉树:

class TreeNode():
    def __init__(self,x):
        self.val = x
        self.left = None
        self.right = None
def method (first,mid):
    if len(first) == 0:
        return None
    if len(first) == 1:
        root = TreeNode(first[0])
        return root
    else:
        root = TreeNode(first[0])
        for i in range(len(mid)):
            if mid[i] == first[0]:
                break
        root.left = method(first[1:i+1],mid[0:i])
        root.right = method(first[i+1:],mid[i+1:])
        return root   

first = [1,3,7,9,5,8,6]
mid = [7,3,9,1,8,5,6]     
root = method(first,mid)
print(root.right.val)
print(root.left.val)
